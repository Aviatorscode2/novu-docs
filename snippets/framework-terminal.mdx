export const FrameworkTerminal = () => {
  if (typeof document === "undefined") {
    return null;
  } else {
    const clientComponent = document.getElementById("nv-framework-terminal-snippet");
    if (clientComponent) {
      clientComponent.innerHTML = '<nv-framework-terminal></nv-framework-terminal>';
    }

    return <div id="nv-framework-terminal-snippet"></div>

}
}
