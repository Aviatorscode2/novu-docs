In this guide, we will add a Novu [Bridge Endpoint](/concepts/endpoint) to a Remix application and send our first test workflow.

<Steps>
    <LocalStudio />
    <PackagesStep />

    <Step title="Add a Novu API Endpoint">
        This guide is based on Remix Offical [Quick Start](https://remix.run/docs/en/main/start/quickstart).

        <CodeGroup>
            ```typescript  app/routes/api.novu.ts 
            import { serve } from "@novu/framework/remix";
            import { client, testWorkflow } from "~/novu/workflows";
            
            const handler = serve({
                client: client,
                workflows: [testWorkflow]
            });
            
            export { handler as action, handler as loader };
            ```
        </CodeGroup>
    </Step>

    <SecretStep />
    <WorkflowStep path="app/novu/workflows.ts" />

    <Step title="Start your application">
        To start your Remix server with the Novu Endpoint configured, run the following command:

        ```bash
        cd my-novu-app && npm run dev
        ```

        Remix application default port is running `5173` port if you used another port, use that instead, and restart the `npx novu dev` command with the port:

        ```bash
        npx novu@latest dev --port <YOUR_REMIX_PORT>
        ```
    </Step>
    <TestStep framework="Remix" />
    <DeployApp />

</Steps>
