## App Router

<CodeGroup>
```ts app/api/echo/route.ts
import { serve } from "@novu/echo/next";
import { echo } from "../../echo/client";

export const { GET, POST, PUT } = serve({ client: echo });
```

</CodeGroup>

## Pages Router

<CodeGroup>
    ```ts pages/api/echo.ts
    import { serve } from "@novu/echo/next";
    import { echo } from '../../echo/client';

    export default serve({ client: echo });
    ```
</CodeGroup>
